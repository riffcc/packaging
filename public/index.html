Public Debian Repository

Current distribution target: Debian 13 amd64.

# Add the signing key
curl -fsSL https://repos.riff.cc/key.asc | sudo gpg --dearmor -o /usr/share/keyrings/rifflabs.gpg

# Add the repository
echo "deb [arch=amd64 signed-by=/usr/share/keyrings/rifflabs.gpg] https://repos.riff.cc/debian trixie main" | sudo tee /etc/apt/sources.list.d/rifflabs.list
